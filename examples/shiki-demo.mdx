# Shiki Code Highlighting Examples

This page demonstrates the various code highlighting features available with Shiki.

## Basic Syntax Highlighting

JavaScript with automatic syntax highlighting:

```js
function greet(name) {
  console.log(`Hello, ${name}!`);
  return `Welcome, ${name}`;
}

// Call the function
const message = greet("World");
```

## Line Highlighting

The function call looks like `greet("World"){:js}` and returns a message.

```js
function add(a, b) {
  // This line is highlighted
  const result = a + b;
  return result;
  // These lines are also highlighted
  // Because we specified them in the range
}
```

## Diff Syntax

Show changes with diff syntax (+ for additions, - for removals):

```diff
function calculateTotal(items) {
-  return items.reduce((acc, item) => acc + item.price, 0);
+  return items
+    .filter(item => !item.deleted)
+    .reduce((acc, item) => acc + item.price, 0);
}
```

```ts
function greet(name) {
  console.log(`hewwo`); // [!code --]
  console.log(`hello`); // [!code ++]
}
```

## Focus Ranges

Focus on specific regions of code:

```js
import { useState, useEffect } from "react";

function UserProfile({ userId }) {
  // #region setup
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  // #endregion setup

  // #region data fetching
  useEffect(() => {
    async function fetchUser() {
      try {
        setLoading(true);
        const response = await fetch(`/api/users/${userId}`);
        const data = await response.json();
        setUser(data);
      } catch (err) {
        setError(err.message);
      } finally {
        setLoading(false);
      }
    }

    fetchUser();
  }, [userId]);
  // #endregion data fetching

  // Render logic here...
}
```

## Word Highlighting

Highlight specific words or phrases:

```jsx
function Counter() {
  // [!highlight: useState]
  const [count, setCount] = useState(0);

  return (
    <div>
      {/* [!highlight: onClick] */}
      <button onClick={() => setCount(count + 1)}>Increment</button>
      <p>Current count: {count}</p>
    </div>
  );
}
```

## TypeScript Example

```ts
interface User {
  id: number;
  name: string;
  email: string;
  role: "admin" | "user" | "guest";
}

function getUserDisplayName(user: User): string {
  return `${user.name} (${user.role})`;
}

// Example usage
const adminUser: User = {
  id: 1,
  name: "Admin User",
  email: "admin@example.com",
  role: "admin",
};

console.log(getUserDisplayName(adminUser)); // Outputs: "Admin User (admin)"
```
